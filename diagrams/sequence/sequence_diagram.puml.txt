@startuml Complete Nutrition Tracker Sequence Diagram
!theme vibrant

' Actors and Participants
actor User
participant TrackerApp
participant RecipeManager
participant NutritionTracker
participant RecipeFactory
participant ProductDAO
participant UserDAO
participant DatabaseConnection
participant ConsoleNutritionObserver
participant Recipe
participant Ingredient
participant NutritionInfo

' User Management Sequences
group User Management
    ' Registration
    User -> TrackerApp: register(username, password, height, weight)
    activate TrackerApp
    TrackerApp -> UserDAO: register(user)
    activate UserDAO
    UserDAO -> DatabaseConnection: getConnection()
    DatabaseConnection --> UserDAO: connection
    UserDAO -> DatabaseConnection: executeQuery()
    DatabaseConnection --> UserDAO: result
    UserDAO --> TrackerApp: registeredUser
    deactivate UserDAO
    TrackerApp --> User: registrationSuccess
    deactivate TrackerApp

    ' Login
    User -> TrackerApp: login(username, password)
    activate TrackerApp
    TrackerApp -> UserDAO: login(username, password)
    activate UserDAO
    UserDAO -> DatabaseConnection: validateCredentials()
    DatabaseConnection --> UserDAO: validationResult
    UserDAO --> TrackerApp: loggedInUser
    deactivate UserDAO
    TrackerApp --> User: loginSuccess
    deactivate TrackerApp

    ' Update User Profile
    User -> TrackerApp: updateProfile(height, weight)
    activate TrackerApp
    TrackerApp -> UserDAO: updateUser(user)
    activate UserDAO
    UserDAO -> DatabaseConnection: executeUpdate()
    DatabaseConnection --> UserDAO: updateResult
    UserDAO --> TrackerApp: updateSuccess
    deactivate UserDAO
    TrackerApp --> User: profileUpdated
    deactivate TrackerApp
end

' Recipe Management Sequences
group Recipe Management
    ' Create Recipe
    User -> TrackerApp: createRecipe(name, servingSize, servingUnit)
    activate TrackerApp
    TrackerApp -> RecipeFactory: createBasicRecipe()
    activate RecipeFactory
    RecipeFactory -> Recipe: new Recipe()
    Recipe --> RecipeFactory: newRecipe
    RecipeFactory --> TrackerApp: recipe
    deactivate RecipeFactory

    ' Add Ingredients to Recipe
    User -> TrackerApp: addIngredient(recipe, ingredientDetails)
    TrackerApp -> RecipeFactory: createIngredient()
    activate RecipeFactory
    RecipeFactory -> Ingredient: new Ingredient()
    Ingredient --> RecipeFactory: newIngredient
    RecipeFactory --> TrackerApp: ingredient
    deactivate RecipeFactory
    TrackerApp -> Recipe: addIngredient(ingredient)
    Recipe --> TrackerApp: updateSuccess

    ' Calculate Recipe Nutrition
    TrackerApp -> Recipe: calculateNutrition()
    activate Recipe
    Recipe -> NutritionInfo: calculateTotal()
    NutritionInfo --> Recipe: totalNutrition
    Recipe --> TrackerApp: nutritionInfo
    deactivate Recipe

    ' Save Recipe
    TrackerApp -> RecipeManager: saveRecipe(recipe)
    activate RecipeManager
    RecipeManager --> TrackerApp: savedRecipe
    deactivate RecipeManager
    TrackerApp --> User: recipeCreated
    deactivate TrackerApp
end

' Product Management Sequences
group Product Management
    ' Add Product
    User -> TrackerApp: addProduct(productDetails)
    activate TrackerApp
    TrackerApp -> ProductDAO: addProduct(product)
    activate ProductDAO
    ProductDAO -> DatabaseConnection: executeInsert()
    DatabaseConnection --> ProductDAO: insertResult
    ProductDAO --> TrackerApp: savedProduct
    deactivate ProductDAO

    ' Update Product
    User -> TrackerApp: updateProduct(productId, newDetails)
    TrackerApp -> ProductDAO: updateProduct(product)
    activate ProductDAO
    ProductDAO -> DatabaseConnection: executeUpdate()
    DatabaseConnection --> ProductDAO: updateResult
    ProductDAO --> TrackerApp: updateSuccess
    deactivate ProductDAO

    ' Delete Product
    User -> TrackerApp: deleteProduct(productId)
    TrackerApp -> ProductDAO: deleteProduct(id)
    activate ProductDAO
    ProductDAO -> DatabaseConnection: executeDelete()
    DatabaseConnection --> ProductDAO: deleteResult
    ProductDAO --> TrackerApp: deleteSuccess
    deactivate ProductDAO

    ' Search Products
    User -> TrackerApp: searchProducts(keyword)
    TrackerApp -> ProductDAO: searchProducts(keyword)
    activate ProductDAO
    ProductDAO -> DatabaseConnection: executeSearch()
    DatabaseConnection --> ProductDAO: searchResults
    ProductDAO --> TrackerApp: matchingProducts
    deactivate ProductDAO
    TrackerApp --> User: displayProducts
    deactivate TrackerApp
end

' Nutrition Tracking Sequences
group Nutrition Tracking
    ' Track Daily Nutrition
    User -> TrackerApp: trackDailyNutrition()
    activate TrackerApp
    TrackerApp -> NutritionTracker: calculateDailyNutrition()
    activate NutritionTracker
    NutritionTracker -> NutritionInfo: aggregateNutrition()
    NutritionInfo --> NutritionTracker: dailyTotal

    ' Notify Observers
    NutritionTracker -> ConsoleNutritionObserver: update(nutritionInfo)
    activate ConsoleNutritionObserver
    ConsoleNutritionObserver --> User: displayNutritionSummary
    deactivate ConsoleNutritionObserver
    NutritionTracker --> TrackerApp: trackingComplete
    deactivate NutritionTracker

    ' Generate Nutrition Report
    User -> TrackerApp: generateNutritionReport()
    TrackerApp -> NutritionTracker: generateReport()
    activate NutritionTracker
    NutritionTracker -> ConsoleNutritionObserver: displayReport()
    ConsoleNutritionObserver --> User: showDetailedReport
    NutritionTracker --> TrackerApp: reportGenerated
    deactivate NutritionTracker
    TrackerApp --> User: reportDisplayed
    deactivate TrackerApp
end

' Recipe Search and Filter Sequences
group Recipe Search and Filtering
    ' Search Recipes
    User -> TrackerApp: searchRecipes(criteria)
    activate TrackerApp
    TrackerApp -> RecipeManager: searchRecipes(criteria)
    activate RecipeManager
    RecipeManager --> TrackerApp: matchingRecipes
    deactivate RecipeManager

    ' Filter Recipes
    User -> TrackerApp: filterRecipes(nutritionCriteria)
    TrackerApp -> RecipeManager: filterByNutrition(criteria)
    activate RecipeManager
    RecipeManager --> TrackerApp: filteredRecipes
    deactivate RecipeManager
    TrackerApp --> User: displayFilteredRecipes
    deactivate TrackerApp
end
