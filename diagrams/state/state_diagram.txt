@startuml Nutrition Tracker State Diagram
skinparam StateBackgroundColor LightBlue
skinparam StateBorderColor DarkBlue
skinparam StateStartColor Green
skinparam StateEndColor Red
skinparam StateFontStyle bold
skinparam ArrowColor DarkBlue
skinparam ArrowThickness 2

scale 0.9

[*] --> SystemInitialization

state SystemInitialization {
    [*] --> DatabaseConnection
    DatabaseConnection --> ApplicationStart : Connection Success
    DatabaseConnection --> SystemError : Connection Failed
    SystemError --> DatabaseConnection : Retry
}

state "User Authentication" as Auth {
    [*] --> NotLoggedIn
    
    state NotLoggedIn {
        [*] --> ShowLoginScreen
        ShowLoginScreen --> ValidatingCredentials : Enter Credentials
        ValidatingCredentials --> LoggedIn : Valid
        ValidatingCredentials --> ShowLoginScreen : Invalid
    }
    
    state LoggedIn {
        [*] --> MainDashboard
    }
}

state "Main System" as MainSystem {
    state MainDashboard {
        [*] --> MenuSelection
        MenuSelection --> UserProfileState : Select Profile
        MenuSelection --> RecipeState : Select Recipes
        MenuSelection --> ProductState : Select Products
        MenuSelection --> NutritionState : Select Nutrition
    }

    state "Recipe Management" as RecipeState {
        state "Recipe Operations" as RecipeOps {
            [*] --> ViewingRecipes
            ViewingRecipes --> CreatingRecipe : New Recipe
            ViewingRecipes --> EditingRecipe : Edit
            ViewingRecipes --> DeletingRecipe : Delete
        }

        state CreatingRecipe {
            [*] --> EnteringRecipeInfo
            EnteringRecipeInfo --> AddingIngredients : Next
            AddingIngredients --> CalculatingNutrition : Ingredients Added
            CalculatingNutrition --> SavingRecipe : Calculate Complete
            SavingRecipe --> ViewingRecipes : Save Success
        }
    }

    state "Product Management" as ProductState {
        state "Product Operations" as ProductOps {
            [*] --> ViewingProducts
            ViewingProducts --> AddingProduct : New Product
            ViewingProducts --> EditingProduct : Edit
            ViewingProducts --> DeletingProduct : Delete
        }

        state AddingProduct {
            [*] --> EnteringProductInfo
            EnteringProductInfo --> ValidatingProduct : Submit
            ValidatingProduct --> SavingProduct : Valid
            SavingProduct --> ViewingProducts : Save Success
        }
    }

    state "Nutrition Tracking" as NutritionState {
        state "Tracking Operations" as TrackingOps {
            [*] --> DailyTracking
            DailyTracking --> ViewingHistory : View History
            DailyTracking --> SettingGoals : Set Goals
            DailyTracking --> GeneratingReports : Generate Report
        }

        state DailyTracking {
            [*] --> RecordingIntake
            RecordingIntake --> CalculatingDaily : Add Food
            CalculatingDaily --> UpdatingProgress : Calculate
            UpdatingProgress --> NotifyingObservers : Update
            NotifyingObservers --> RecordingIntake : Complete
        }
    }

    state "User Profile" as UserProfileState {
        [*] --> ViewingProfile
        ViewingProfile --> EditingProfile : Edit
        EditingProfile --> SavingProfile : Save
        SavingProfile --> ViewingProfile : Success
    }
}

' Main Transitions
SystemInitialization --> Auth : System Ready
Auth --> MainSystem : Authentication Success
MainSystem --> Auth : Logout

' Error Handling
state SystemError {
    [*] --> ErrorLogging
    ErrorLogging --> ErrorNotification
    ErrorNotification --> ErrorRecovery
    ErrorRecovery --> [*]
}

' Return to Dashboard
RecipeState --> MainDashboard : Back
ProductState --> MainDashboard : Back
NutritionState --> MainDashboard : Back
UserProfileState --> MainDashboard : Back

' Observer Pattern States
state "Observer System" as ObserverSystem {
    [*] --> ObserverIdle
    ObserverIdle --> NotifyingObservers : Change Detected
    NotifyingObservers --> ObserversUpdated : Notification Complete
    ObserversUpdated --> ObserverIdle : Reset
}

note right of SystemInitialization
  System startup and
  database connection
end note

note right of Auth
  User authentication
  and session management
end note

note right of MainSystem
  Core system functionality
  and user operations
end note

note right of ObserverSystem
  Nutrition tracking
  notification system
end note

@enduml
